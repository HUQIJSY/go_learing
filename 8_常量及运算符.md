```
package main

import "fmt"

// import "strconv"

// import "math"

// 为类型声明一个别名
type (
	byte int8
	rune int32
	文本   string
)

const g, h, j = 1, 2, 4
const a int = 1
const b = "A"
const (
	c = a
	d = a + 1
	e = a + 2
)

func main() {
	fmt.Println(a)
	fmt.Println(b)
	fmt.Println(c)
	fmt.Println(e)
	fmt.Println(g, h, j)
}
```
